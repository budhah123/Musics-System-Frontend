# Firebase Storage Rules for Audio Streaming
# Copy these rules to your Firebase Console > Storage > Rules

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow public read access to audio and image files
    match /{allPaths=**} {
      // Allow read access to all files (for public streaming)
      allow read: if true;
      
      // Allow write access only to authenticated users
      allow write: if request.auth != null;
      
      // Allow delete access only to authenticated users
      allow delete: if request.auth != null;
    }
  }
}
